Sub Print()
    Dim Sht As Variant
     
    'ブックの全シートに対する処理
    For Each Sht In Worksheets
       'シート名を取得し、特定シート名以外は印刷を実行
       Select Case Sht.Name
           Case Is <> "改定履歴"
              Sht.PrintOut Preview:=True
       End Select
    Next
    '**シート毎に判定するループ
    '各シートのプリントスプールが別々になるので、出力時に別のPCから受けた印刷物と混ざる可能性あり
    '念のためプレビューあり
End Sub
